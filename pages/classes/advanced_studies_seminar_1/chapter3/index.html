自動微分 

import torch
import torch.nn as nn
import numpy as np

x = torch.tensor([2.0], requires_grad=True) # x=2
print(x)

#出力
tensor([2.], requires_grad=True)

y = x * x + 2 * x + 1 #x^2 + 2*x + 1
print(y)

#出力
tensor([9.], grad_fn=<AddBackward0>)

y.backward() #微分

print(f"dy/dx = {x.grad}") # 2*x + 2

#出力
dy/dx = tensor([6.])
