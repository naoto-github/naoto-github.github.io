Monacaでアプリ開発② ONSEN UIの利用

Onsen UI

プロジェクトの作成

新規にプロジェクトを作成します． テンプレートは Onsen UI V2 Vue Splitter を選択しましょう．

プロジェクト名に「Monacaでアプリ開発②」，説明に「Onsen UIの利用」を設定しましょう．

テンプレートの確認

App.vue

App.vueを確認しましょう．

HTML

<template>

  <v-ons-splitter>
	<!-- 左側に配置されるメニュー -->
    <v-ons-splitter-side
      swipeable width="150px" collapse="" side="left"
      :open.sync="openSide"
    >
      <v-ons-page>
        <v-ons-list>
          <v-ons-list-header>Menu</v-ons-list-header>
		  
		  <!-- v-forディレクティブでメニューの項目を表示 -->
          <v-ons-list-item v-for="page in pages" :key="page" tappable modifier="chevron"
            @click="currentPage = page; openSide = false"
          >
            <div class="center">{{ page }}</div>
          </v-ons-list-item>
        </v-ons-list>
      </v-ons-page>
    </v-ons-splitter-side>

	<!-- 右側に配置されるメインコンテンツ -->
    <v-ons-splitter-content>
      <div :is="currentPage" :toggle-menu="() => openSide = !openSide"></div>
    </v-ons-splitter-content>
  </v-ons-splitter>
</template>

<script>
	<!-- JavaScriptのコード -->
</script>

JavaScript

import home from './components/homePage'
import news from './components/newsPage'
import settings from './components/settingsPage'
export default {
  data() {
    return {
      currentPage: 'home', // 最初のページはhome
      pages: ['home', 'news', 'settings'], // メニューに表示される項目
      openSide: false // 最初はメニューを閉じる
    };
  },
  components: {
    home, // homeコンポーネントの登録
    news, // newsコンポーネントの登録
    settings // settingsコンポーネントの登録
  }
}
